Lemma Sextet_from_nat_impl_Sextet_from_nat_safe : ∀ n v H,
  Sextet_from_nat n = Some v →
  Sextet_from_nat_safe n H = v.

  intros v;
  destruct v;
  simpl in *;
  repeat (break_match; simpl in *; try simple congruence 3; eauto); eauto.
