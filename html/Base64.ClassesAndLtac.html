Ltac clean :=
  match goal with
    | [ H : ?X = ?X |- _ ] ⇒ clear H
  end.

Ltac subst_max :=
  repeat clean;
  repeat match goal with
          | [ H : ?X = _ |- _ ] ⇒ subst X
          | [H : _ = ?X |- _] ⇒ subst X
        end.

Ltac find_rew :=
  subst_max;
  match goal with
  | [ H : ?X _ _ _ _ = _, H' : ?X _ _ _ _ = _ |- _ ] ⇒ rewrite H in H'
  | [ H : ?X = _, H' : ?X = _ |- _ ] ⇒ rewrite H in H'
  | [ H : ?X = _, H' : context [ ?X ] |- _ ] ⇒ rewrite H in H'
  | [ H : ?X = _ |- context [ ?X ] ] ⇒ rewrite H
  end.

Ltac find_rev_rew :=
  subst_max;
  match goal with
    | [ H : _ = ?X _ _ _ _, H' : ?X _ _ _ _ = _ |- _ ] ⇒ rewrite <- H in H'
    | [ H : _ = ?X, H' : context [ ?X ] |- _ ] ⇒ rewrite <- H in H'
    | [ H : _ = ?X |- context [ ?X ] ] ⇒ rewrite <- H
  end.

Definition string_ind3 :=
  fun (P : string → Prop)
    (f0 : P EmptyString)
    (f1 : ∀ a, P (String a EmptyString))
    (f2 : ∀ a1 a2, P (String a1 (String a2 EmptyString)))
    (f3 : ∀ a1 a2 a3 rest,
      P rest →
      P (String a1 (String a2 (String a3 rest)))) ⇒
  fix F (s1 : string) : P s1 :=
  match s1 with
  | EmptyString ⇒ f0
  | String a1 s2 ⇒
    match s2 with
    | EmptyString ⇒ f1 a1
    | String a2 s3 ⇒
      match s3 with
      | EmptyString ⇒ f2 a1 a2
      | String a3 s4 ⇒ f3 a1 a2 a3 s4 (F s4)
      end
    end
  end.

  end.

Class Encodable (A B : Type) := {
  encode : A → B;
  decode : B → option A;
  invol : ∀ a, decode (encode a) = Some a
}.

Class StrictEncodable (A B : Type) := {
  strict_encode : A → B;
  strict_decode : B → A;
  strict_invol : ∀ a, strict_decode (strict_encode a) = a
}.

Theorem strict_encode_inj {A B} `{StrictEncodable A B} : ∀ a1 a2,
  strict_encode a1 = strict_encode a2 → a1 = a2.

Global Instance StrictEncodable_refl {A : Type} : StrictEncodable A A := {|
  strict_encode := fun x ⇒ x;
  strict_decode := fun x ⇒ x;
  strict_invol := fun x ⇒ eq_refl
|}.

Global Instance StrictEncodable_trans {A B C : Type}
  `{StrictEncodable A B, StrictEncodable B C} : StrictEncodable A C.
Defined.

Global Instance Encodable_from_strict (A B : Type) `{StrictEncodable A B} : Encodable A B.

Class EqClass (A : Type) := {
  eqb : A → A → bool;
  eqb_eq : ∀ x y, eqb x y = true ↔ x = y
}.

Class DecEq (A : Type) := {
  decEq : ∀ x y : A, {x = y} + {x ≠ y}
}.

Lemma eqb_refl : ∀ A `{EqClass A} (a : A),
  eqb a a = true.

Lemma eqb_neq : ∀ A `{EqClass A} (a b : A),
  eqb a b = false ↔ a ≠ b.

Theorem decEq_impl_id_irrel : ∀ A `{DecEq A} (x y : A) (H1 H2 : x = y),
  H1 = H2.

Global Instance DecEq_from_EqClass (A : Type) `{EqClass A} : DecEq A.
Qed.
